	\begin{lemma}[CaractÃĐrisation des distributions de dÃĐrivÃĐe nulle]

	\begin{proof}
		\emph{($\Leftarrow$)} Si $T = C$ (constante), alors $\langle T', \varphi\rangle = -\langle T, \varphi'\rangle = -C\int\varphi' = 0$.
		
		\emph{($\Rightarrow$)} Supposons $T' = 0$, c'est-Ã -dire $\langle T, \varphi'\rangle = 0$ pour toute $\varphi \in \mathcal{D}$.
		
		\textbf{Ãtape 1 (image de la dÃĐrivation).} Montrons que l'image de $\varphi \mapsto \varphi'$ dans $\mathcal{D}$ est prÃĐcisÃĐment
		\[
		\mathcal{D}_0 := \left\{\psi \in \mathcal{D}\;\middle|\; \int_{-\infty}^{+\infty}\psi(x)\,dx = 0\right\}.

	\end{proof}
